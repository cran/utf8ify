Format text

Use utf8_text_*() functions to format text (using Utf8)

library(utf8ify)
text <- "\nHello World"

cat(
  text,
  utf8_text_bold(text),
  utf8_text_bold2(text),
  utf8_text_italic(text),
  utf8_text_bolditalic(text),
  utf8_text_cursive(text),
  utf8_text_gothic(text),
  utf8_text_circle(text)
)
